double angleAtTime(double t)
{
    const double alpha = 6.0;
    const double omega_max = 60.0;
    const double t_max = omega_max / alpha; // 10s

    if (t <= t_max)
    {
        return 0.5 * alpha * t * t; // 3*t^2
    }
    else
    {
        return 0.5 * alpha * t_max * t_max
             + omega_max * (t - t_max);
    }
}
