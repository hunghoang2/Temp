double computeAngleDeg(
    double angular_accel_deg_s2, // gia tốc góc (deg/s^2)
    double dt,                   // thời gian (s)
    double alpha_max = 10.0       // giới hạn gia tốc góc
)
{
    angular_accel_deg_s2 =
        std::clamp(angular_accel_deg_s2, -alpha_max, alpha_max);

    // θ = 1/2 * α * t²
    return 0.5 * angular_accel_deg_s2 * dt * dt;
}
