#include <ctime>
#include <iomanip>
#include <sstream>
#include <string>

std::string utc_to_string_am_pm(time_t utc_ts)
{
    // UTC -> tm
    std::tm utc_tm = *gmtime(&utc_ts);

    // UTC+7 (Việt Nam)
    utc_tm.tm_hour += 7;
    mktime(&utc_tm); // normalize lại ngày/tháng/năm

    std::ostringstream oss;
    oss << std::put_time(&utc_tm, "%d/%m/%Y %I:%M %p");

    return oss.str();
}
