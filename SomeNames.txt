cv::Mat gray, mask;
cv::cvtColor(frame(roi), gray, cv::COLOR_BGR2GRAY);
cv::threshold(gray, mask, 50, 255, cv::THRESH_BINARY);
std::vector<std::vector<cv::Point>> contours;
cv::findContours(mask, contours, cv::RETR_EXTERNAL, cv::CHAIN_APPROX_SIMPLE);

if (!contours.empty()) {
    cv::Rect objectBox = cv::boundingRect(contours[0]);
    roi.x += objectBox.x;
    roi.y += objectBox.y;
    roi.width = objectBox.width;
    roi.height = objectBox.height;
}
