#include <QDateTime>
#include <QString>

QString utcToStringAMPM(qint64 utcSeconds)
{
    QDateTime dt = QDateTime::fromSecsSinceEpoch(
        utcSeconds,
        Qt::UTC
    );

    // UTC+7 (Viá»‡t Nam)
    dt = dt.toTimeZone(QTimeZone("Asia/Ho_Chi_Minh"));

    return dt.toString("dd/MM/yyyy h:mm AP");
}
